    List subnatures for a given nature

def self.list_for_nature(nature)
  subnatures = Subnature.where(nature_id: 0, splat_id: nature.splat.id)
  subnatures = subnatures + Subnature.where(nature_id: nature.id)
  subnatures.collect
end
